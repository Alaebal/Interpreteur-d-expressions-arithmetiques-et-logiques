fonction factorielle(n) faire
    si n <= 1 alors
        retourner 1;
    fin
    retourner n * factorielle(n - 1);
fin

fonction max(a, b) faire
    si a > b alors
        retourner a;
    sinon
        retourner b;
    fin
fin
